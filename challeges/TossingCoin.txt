import {Component} from "react"
import HeadToss from "./components/tossCard/index"
import './App.css'

const tossDetails = [
  {
    id: 1,
    imageUrl: "https://assets.ccbp.in/frontend/react-js/heads-img.png",
    altText: "Head"
  },
  {
    id: 2,
    imageUrl : "https://assets.ccbp.in/frontend/react-js/tails-img.png",
    altText : "Tail"
  }
]

class App extends Component {
  state = {activeCoin:tossDetails[0].id,total: 0,heads: 0,tails: 0}

  onTossCoin = () => { 
    let {heads,tails} = this.state
    const updateToss = Math.ceil(Math.random()*tossDetails.length);
    const updateHeads = updateToss === 1 ? heads+=1 : heads;
    const updateTails = updateToss === 2 ? tails+=1 : tails;
    this.setState((prevValue) => ({total:prevValue.total+1,heads:updateHeads,tails:updateTails,activeCoin:updateToss }))
  }
  tossedCoin = () => {
    const {activeCoin} = this.state
    const tossedValue = tossDetails.filter(eachCoin => (eachCoin.id===activeCoin))
    return tossedValue
  }
  render(){
    const {total,heads,tails} = this.state
    const tossedValue = this.tossedCoin()
    return(
      <div className= "bg-container">
        <div className="toss-card">
          <h1>Coin Toss Game</h1>
          <p>Heads or Tails</p>
          {tossedValue.map(eachItem => (<HeadToss key={eachItem.id} tossValue={eachItem}/>))}
          <button onClick={this.onTossCoin}>Toss Coin</button>
          <div className="card-count">
          <p className="para">Total: {total}</p>
          <p className="para">Heads: {heads}</p>
          <p className="para">Tails: {tails}</p>
          </div>
        </div>
      </div>
    )
  }
}

export default App
