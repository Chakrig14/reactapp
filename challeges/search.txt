class DisplayName extends Component {
  state = {searchInput: '',inputSearchDetails: destinationPlaces}
  
  onChangeSearch = event => {
    (this.setState({searchInput:event.target.value}))
  }
  render(){
    const {searchInput,inputSearchDetails} = this.state;
    const searchInputResults = inputSearchDetails.filter((eachResult) => {
      let searchName = eachResult.name;
      let lowerCaseName= searchName.toLowerCase();
      return (
        lowerCaseName.includes(searchInput)
      )
    })
    return (
      <div className="bg-container">
        <div className="top-section">
        <h1 className="heading">Destination Search</h1>
        <input type="search" placeholder="Search a Destination" className="search" onChange={this.onChangeSearch}/>
        </div>
        <div className="card">
          {searchInputResults.map((eachItem) => (
            <Destination destinationPlace={eachItem} key={eachItem.id}/>
          ))}
        </div>
      </div>
    )  
  }
}
export default DisplayName;
